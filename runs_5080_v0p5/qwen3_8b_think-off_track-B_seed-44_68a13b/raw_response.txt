#!/usr/bin/env bash
set -euo pipefail

mkdir -p results

for sample in M117-bl M117-ch M117C1-bl M117C1-ch; do
    if [[ -f results/${sample}.bam && -f results/${sample}.bam.bai && -f results/${sample}.vcf.gz && -f results/${sample}.vcf.gz.tbi ]]; then
        continue
    fi

    # Align with bwa
    bwa mem data/ref/chrM.fa data/raw/${sample}_1.fq.gz data/raw/${sample}_2.fq.gz | samtools view -S
