#!/usr/bin/env bash
set -euo pipefail

THREADS=4
REF="data/ref/chrM.fa"
RAW="data/raw"
RESULTS="results"

SAMPLES=(M117-bl M117-ch M117C1-bl M117C1-ch)

mkdir -p "${RESULTS}"

    bwa index "${REF}"
fi

if [[ ! -f "${REF}.fai" ]]; then
    samtools faidx "${REF}"
fi

    R1="${RAW}/${SAMPLE}_1.fq.gz"
