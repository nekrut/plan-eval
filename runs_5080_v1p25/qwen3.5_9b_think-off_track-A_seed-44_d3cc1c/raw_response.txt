#!/usr/bin/env bash
set -euo pipefail

THREADS=4
SAMPLES="M117-bl M117-ch M117C1-bl M117C1-ch"
REF="data/ref/chrM.fa"
RAW="data/raw"
RES="results"

mkdir -p "$RES"

REF_IDX="$REF.fai"
REF_BWA="chrM.amb"

for s in $SAMPLES; do
    [ -f "$RES/${s}.vcf.gz.tbi" ] && continue
    [ -f "$REF_IDX" ] || samtools faidx "$REF"
    [ -f "$REF_BWA" ] || bwa index "$REF"
    bwa mem -t $THREADS -R "@RG\tID:${s
