#!/usr/bin/env bash
set -euo pipefail
THREADS=4
SAMPLES="M117-bl M117-ch M117C1-bl M117C1-ch"
REF="data/ref/chrM.fa"
RAW="data/raw"
RES="results"

mkdir -p "$RES"

REF_IDX="$REF.fai"
REF_BWA="chrM"

if [[ ! -f "$REF_IDX" ]]; then
    samtools faidx "$REF"
fi

if [[ ! -f "$REF_BWA.amb" ]]; then
    bwa index "$REF"
fi

for SAMPLE in $SAMPLES; do
    F1="$RAW/${SAMPLE}_1.fq.gz"
    F2="$RAW/${SAMPLE
