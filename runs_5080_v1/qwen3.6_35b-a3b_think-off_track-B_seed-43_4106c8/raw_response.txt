#!/usr/bin/env bash
set -euo pipefail

# Create output directory
mkdir -p results

# Reference files
REF=data/ref/chrM.fa
REF_BWA=data/ref/chrM.fa.bwt
REF_SAMTOOLS=data/ref/chrM.fa.fai

# Index reference if not present
if [ ! -f "$REF_BWA" ]; then
    bwa index "$REF"
fi

if [ ! -f "$REF_SAMTOOLS" ]; then
    samtools faidx "$REF"
fi
